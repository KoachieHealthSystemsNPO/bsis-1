CREATE TABLE User (
 userId INT PRIMARY KEY AUTO_INCREMENT,
 username VARCHAR(32),
 password VARCHAR(32),
 type VARCHAR(32),
 name VARCHAR(250),
 contactNumber VARCHAR(50),
 emailId VARCHAR(50),
 isDeleted INT Default 0,
 comments VARCHAR(250)
) ENGINE=InnoDB;


CREATE TABLE Donor (
 donorId INT PRIMARY KEY AUTO_INCREMENT,
 donorNumber VARCHAR(32),
 firstName VARCHAR(32),
 lastName VARCHAR(32),
 gender VARCHAR(32),
 bloodType VARCHAR(32),
 birthDate DATE,
 address VARCHAR(64),
 age INT,
 isDeleted INT Default 0,
 comments VARCHAR(250)
) ENGINE=InnoDB;


CREATE TABLE Collection (
 collectionId INT PRIMARY KEY AUTO_INCREMENT,
 collectionNumber VARCHAR(32),
 centerId INT,
 siteId INT,
 dateCollected DATE,
 donorNumber VARCHAR(32),
 donorType VARCHAR(32),
 sampleNumber INT,
 shippingNumber INT,
 abo VARCHAR(250),
 rhd VARCHAR(250),
 isDeleted INT Default 0,
 comments VARCHAR(250)
 )ENGINE=InnoDB;

CREATE TABLE TestResult (
 testResultId INT PRIMARY KEY AUTO_INCREMENT,
 collectionNumber VARCHAR(32),
 dateCollected DATE,
 dateTested DATE,
 hiv VARCHAR(250),
 hbv VARCHAR(250),
 hcv VARCHAR(250),
 syphilis VARCHAR(250),
 abo VARCHAR(250),
 rhd VARCHAR(250),
 isDeleted INT Default 0,
 comments VARCHAR(250)
 )ENGINE=InnoDB;


 CREATE TABLE Product (
 productId INT PRIMARY KEY AUTO_INCREMENT,
 productNumber VARCHAR(32),
 collectionNumber VARCHAR(32),
 dateCollected DATE,
 type VARCHAR(32),
 abo VARCHAR(250),
 rhd VARCHAR(250),
 isIssued INT Default 0,
 isDeleted INT Default 0,
 comments VARCHAR(250)
 )ENGINE=InnoDB;

 CREATE TABLE LocationType (
 locationTypeId INT PRIMARY KEY AUTO_INCREMENT,
 name VARCHAR(32),
 isDeleted INT Default 0,
 comments VARCHAR(250)
) ENGINE=InnoDB;

CREATE TABLE Location (
 locationId INT PRIMARY KEY AUTO_INCREMENT,
 name VARCHAR(32),
 type INT,
 isCenter INT,
 isCollectionSite INT,
 isUsageSite INT,
 isMobileSite INT,
 isDeleted INT Default 0,
 comments VARCHAR(250),
 FOREIGN KEY (type) REFERENCES LocationType (locationTypeId)
) ENGINE=InnoDB;

CREATE TABLE ReportConfig (
 reportType VARCHAR(32) PRIMARY KEY,
 fieldNames TEXT
) ENGINE=InnoDB;


CREATE TABLE DisplayNames (
 formType VARCHAR(32) PRIMARY KEY,
 fieldNames TEXT
) ENGINE=InnoDB;

CREATE TABLE Request (
 requestId INT PRIMARY KEY AUTO_INCREMENT,
 requestNumber VARCHAR(32),
 dateRequested DATE,
 siteId INT,
 abo VARCHAR(250),
 rhd VARCHAR(250),
 productType VARCHAR(32),
 quantity INT,
 dateRequired DATE,
 status VARCHAR(250),
 isDeleted INT Default 0,
 comments VARCHAR(250)
 )ENGINE=InnoDB;

 CREATE TABLE Issue (
 issueId INT PRIMARY KEY AUTO_INCREMENT,
 productNumber VARCHAR(32),
 dateIssued DATE,
 siteId INT,
 isDeleted INT Default 0,
 comments VARCHAR(250)
 )ENGINE=InnoDB;


 CREATE TABLE ProductUsage(
 usageId INT PRIMARY KEY AUTO_INCREMENT,
 productNumber VARCHAR(32),
 dateUsed DATE,
 hospital VARCHAR(250),
 ward VARCHAR(250),
 useIndication VARCHAR(200),
 isDeleted INT Default 0,
 comments VARCHAR(250)
 )ENGINE=InnoDB;

CREATE TABLE RecordFieldsConfig (
recordType VARCHAR(32) PRIMARY KEY,
fieldNames TEXT
) ENGINE=InnoDB;

CREATE TABLE Feedback (
feedbackId INT PRIMARY KEY AUTO_INCREMENT,
comment VARCHAR(250),
datetime TIMESTAMP DEFAULT CURRENT_TIMESTAMP
) ENGINE=InnoDB;
